qingqiu ask "status"

qingqiu memory get user_name

qingqiu task add "写文档"

qingqiu status
qingqiu config show
qingqiu llm test ollama

# Voice（需麦克风）

uv run python -m qingqiu.daemon   # 127.0.0.1:7788

uv run python -m qingqiu.ui       # 127.0.0.1:7789 → http://localhost:7789
